git init / rm -rf .git

git add -nombre archivo- (to the staging area, area de ensayo, manda a hacer seguimiento)

git commit -m "Descripcion" (to the local repository, crea copia de respaldo del archivo tal com esta en el momento)

git commit -am (commit + add)

git status -s (para ver si git realmente estan sinedo sometidos a seguimiento)

git log --oneline

git reset --hard -numero-

git branch (veo los branches)

git branch -nombre- (la creo)

git branch -d -nombre- (delete)

git checkout (para moverte a trabajar en otra branch)

git merge -nombre branch-
